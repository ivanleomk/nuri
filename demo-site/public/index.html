index.zig Generated

src/app/index.zig 
pub const meta = .{
    .title = "Home",
};

pub fn render() !mer.Node {
    return mer.html(.{}, .{
        mer.h1(.{}, "Welcome to Nuri"),
        mer.p(.{}, .{
            "This is your first page with ",
            "", mer.strong(.{}, "bold"), "",
            " and ",
            "", mer.em(.{}, "italic"), "",
            " text.",
        }),
        mer.p(.{}, .{
            "Check out ",
            "", mer.a(.{.href = "/about"}, "this link"), "",
            ".",
        }),
    });
}
